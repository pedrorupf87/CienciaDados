### **1. Atualizar Freight com aumento de 10% para pedidos com valor total > $500**
```sql
-- Criação de uma tabela de log para auditoria
CREATE TABLE LogFreightUpdate (
    OrderID INT,
    OldFreight DECIMAL(10,2),
    NewFreight DECIMAL(10,2),
    UpdateDate DATETIME
);

DECLARE @OrderID INT, @Freight DECIMAL(10,2), @TotalValue DECIMAL(10,2);

DECLARE freight_cursor CURSOR FOR
    SELECT o.OrderID, o.Freight, SUM(od.UnitPrice * od.Quantity * (1 - od.Discount)) AS TotalValue
    FROM Orders o
    JOIN [Order Details] od ON o.OrderID = od.OrderID
    GROUP BY o.OrderID, o.Freight
    HAVING SUM(od.UnitPrice * od.Quantity * (1 - od.Discount)) > 500;

OPEN freight_cursor;
FETCH NEXT FROM freight_cursor INTO @OrderID, @Freight, @TotalValue;

WHILE @@FETCH_STATUS = 0
BEGIN
    DECLARE @NewFreight DECIMAL(10,2) = @Freight * 1.10; -- Aumento de 10%
    UPDATE Orders
    SET Freight = @NewFreight
    WHERE OrderID = @OrderID;

    INSERT INTO LogFreightUpdate (OrderID, OldFreight, NewFreight, UpdateDate)
    VALUES (@OrderID, @Freight, @NewFreight, GETDATE());

    FETCH NEXT FROM freight_cursor INTO @OrderID, @Freight, @TotalValue;
END;

CLOSE freight_cursor;
DEALLOCATE freight_cursor;

-- Verificação
SELECT * FROM LogFreightUpdate;
```

### **3. Reduzir preço em 5% para produtos com estoque > 100**
```sql
DECLARE @ProductID INT, @UnitPrice DECIMAL(10,2), @UnitsInStock SMALLINT;

DECLARE product_cursor CURSOR FOR
    SELECT ProductID, UnitPrice, UnitsInStock
    FROM Products
    WHERE UnitsInStock > 100;

OPEN product_cursor;
FETCH NEXT FROM product_cursor INTO @ProductID, @UnitPrice, @UnitsInStock;

WHILE @@FETCH_STATUS = 0
BEGIN
    BEGIN TRY
        DECLARE @NewPrice DECIMAL(10,2) = @UnitPrice * 0.95; -- Redução de 5%
        IF @NewPrice < 0
            THROW 50001, 'Preço não pode ser negativo.', 1;
        
        UPDATE Products
        SET UnitPrice = @NewPrice
        WHERE ProductID = @ProductID;
    END TRY
    BEGIN CATCH
        PRINT 'Erro ao atualizar Produto ' + CAST(@ProductID AS VARCHAR) + ': ' + ERROR_MESSAGE();
    END CATCH;

    FETCH NEXT FROM product_cursor INTO @ProductID, @UnitPrice, @UnitsInStock;
END;

CLOSE product_cursor;
DEALLOCATE product_cursor;
```

### **13. Produtos vendidos acima do preço médio da categoria**
```sql
CREATE TABLE PriceExceptions (
    OrderID INT,
    ProductID INT,
    UnitPrice DECIMAL(10,2),
    AvgCategoryPrice DECIMAL(10,2)
);

DECLARE @OrderID INT, @ProductID INT, @UnitPrice DECIMAL(10,2), @AvgPrice DECIMAL(10,2);

DECLARE price_cursor CURSOR FOR
    SELECT od.OrderID, od.ProductID, od.UnitPrice, 
           AVG(od.UnitPrice) OVER (PARTITION BY p.CategoryID) AS AvgPrice
    FROM [Order Details] od
    JOIN Products p ON od.ProductID = p.ProductID
    WHERE od.UnitPrice > (SELECT AVG(UnitPrice) FROM Products p2 WHERE p2.CategoryID = p.CategoryID);

    VALUES (@OrderID, @ProductID, @UnitPrice, @AvgPrice);
```

### **21. Valor total de vendas por categoria em 1997**
```sql
WITH SalesByCategory AS (
    SELECT 
        c.CategoryID,
        c.CategoryName,
        SUM(od.UnitPrice * od.Quantity * (1 - od.Discount)) AS TotalSales
    FROM Categories c
    JOIN Products p ON c.CategoryID = p.CategoryID
    JOIN [Order Details] od ON p.ProductID = od.ProductID
    JOIN Orders o ON od.OrderID = o.OrderID
    WHERE YEAR(o.OrderDate) = 1997
    GROUP BY c.CategoryID, c.CategoryName
)
SELECT 
    CategoryID,
    CategoryName,
    TotalSales
FROM SalesByCategory
ORDER BY TotalSales DESC;

-- Explicação: A CTE calcula o valor total de vendas com base em Order Details, filtrando pedidos de 1997.
```

### **22. Hierarquia completa de funcionários (recursiva)**
```sql
WITH EmployeeHierarchy AS (
    SELECT 
        EmployeeID,
        FirstName + ' ' + LastName AS EmployeeName,
        ReportsTo,
        1 AS Level
    FROM Employees
    WHERE ReportsTo IS NULL  -- Começa pelo topo (sem chefe)
    UNION ALL
    SELECT 
        e.EmployeeID,
        e.FirstName + ' ' + e.LastName,
        e.ReportsTo,
        eh.Level + 1
    FROM Employees e
    INNER JOIN EmployeeHierarchy eh ON e.ReportsTo = eh.EmployeeID
)
SELECT 
    EmployeeID,
    EmployeeName,
    ReportsTo AS ManagerID,
    Level
FROM EmployeeHierarchy
ORDER BY Level, EmployeeID;
```

### **25. Crescimento mensal de vendas em 1998**
```sql
WITH MonthlySales AS (
    SELECT 
        YEAR(o.OrderDate) AS Year,
        MONTH(o.OrderDate) AS Month,
        SUM(od.UnitPrice * od.Quantity * (1 - od.Discount)) AS TotalSales
    FROM Orders o
    JOIN [Order Details] od ON o.OrderID = od.OrderID
    WHERE YEAR(o.OrderDate) = 1998
    GROUP BY YEAR(o.OrderDate), MONTH(o.OrderDate)
),
SalesGrowth AS (
    SELECT 
        Year,
        Month,
        TotalSales,
        LAG(TotalSales) OVER (ORDER BY Year, Month) AS PreviousMonthSales,
        (TotalSales - LAG(TotalSales) OVER (ORDER BY Year, Month)) / 
            NULLIF(LAG(TotalSales) OVER (ORDER BY Year, Month), 0) * 100 AS GrowthPercent
    FROM MonthlySales
)
SELECT 
    Year,
    Month,
    TotalSales,
    PreviousMonthSales,
    ROUND(GrowthPercent, 2) AS GrowthPercent
FROM SalesGrowth
ORDER BY Year, Month;

-- Explicação: CTEs aninhadas para calcular vendas mensais e o crescimento percentual em relação ao mês anterior.
```

### **26. Produtos mais vendidos por região (top 3)**
```sql
WITH RankedProducts AS (
    SELECT 
        o.ShipRegion,
        p.ProductID,
        p.ProductName,
        SUM(od.Quantity) AS TotalSold,
        ROW_NUMBER() OVER (PARTITION BY o.ShipRegion ORDER BY SUM(od.Quantity) DESC) AS Rank
    FROM Orders o
    JOIN [Order Details] od ON o.OrderID = od.OrderID
    JOIN Products p ON od.ProductID = p.ProductID
    WHERE o.ShipRegion IS NOT NULL
    GROUP BY o.ShipRegion, p.ProductID, p.ProductName
)
SELECT 
    ShipRegion,
    ProductID,
    ProductName,
    TotalSold
FROM RankedProducts
WHERE Rank <= 3
ORDER BY ShipRegion, TotalSold DESC;

-- Explicação: A CTE usa ROW_NUMBER para classificar os produtos mais vendidos por região.
```

```sql
    Profit
FROM OrderProfit
ORDER BY Profit DESC;
```

```sql
CREATE TABLE PriceUpdateLog (
    ProductID INT,
    OldPrice DECIMAL(10,2),
    NewPrice DECIMAL(10,2),
    UpdateDate DATETIME,
    ErrorMessage NVARCHAR(255)
);

BEGIN TRY
    BEGIN TRANSACTION;
    UPDATE Products
    SET UnitPrice = UnitPrice * 1.10  -- Aumento de 10%
    OUTPUT 
        deleted.ProductID,
        deleted.UnitPrice AS OldPrice,
        inserted.UnitPrice AS NewPrice,
        GETDATE(),
        NULL
    INTO PriceUpdateLog (ProductID, OldPrice, NewPrice, UpdateDate, ErrorMessage)
    WHERE UnitPrice > 0;

    IF EXISTS (SELECT 1 FROM Products WHERE UnitPrice < 0)
        THROW 50001, 'Preço negativo detectado após atualização.', 1;

    COMMIT TRANSACTION;
END TRY
BEGIN CATCH
    ROLLBACK TRANSACTION;
    INSERT INTO PriceUpdateLog (ProductID, OldPrice, NewPrice, UpdateDate, ErrorMessage)
    VALUES (NULL, NULL, NULL, GETDATE(), ERROR_MESSAGE());
    PRINT 'Erro: ' + ERROR_MESSAGE();
```

```sql
        IF @NewDiscount > 1
            THROW 50006, 'Desconto maior que 100%.', 1;

        UPDATE [Order Details]
        SET Discount = @NewDiscount
        WHERE OrderID = @OrderID AND ProductID = @ProductID;
    END TRY
    BEGIN CATCH
        INSERT INTO DiscountLimitLog (OrderID, ProductID, OldDiscount, NewDiscount, ErrorMessage, LogDate)
        VALUES (@OrderID, @ProductID, @Discount, @NewDiscount, ERROR_MESSAGE(), GETDATE());
    END CATCH;

    FETCH NEXT FROM discount_cursor INTO @OrderID, @ProductID, @Discount;
END;

CLOSE discount_cursor;
DEALLOCATE discount_cursor;

-- Verificação
SELECT * FROM DiscountLimitLog;
```
