```sql
CREATE PROCEDURE ApplyProgressiveDiscounts
AS
BEGIN
    DECLARE @OrderID INT, @ProductID INT, @Quantity INT;

    DECLARE OrderDetailCursor CURSOR FOR
    SELECT OrderID, ProductID, Quantity FROM [Order Details]
    WHERE Quantity > 5;

    OPEN OrderDetailCursor;
    FETCH NEXT FROM OrderDetailCursor INTO @OrderID, @ProductID, @Quantity;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        UPDATE [Order Details]
        SET Discount = CASE
            WHEN @Quantity BETWEEN 6 AND 10 THEN 0.10
            WHEN @Quantity BETWEEN 11 AND 20 THEN 0.15
            ELSE 0.20
        END
        WHERE OrderID = @OrderID AND ProductID = @ProductID;

        FETCH NEXT FROM OrderDetailCursor INTO @OrderID, @ProductID, @Quantity;
    END;

    CLOSE OrderDetailCursor;
    DEALLOCATE OrderDetailCursor;
END;
```

```sql
CREATE PROCEDURE UpdateFreightByRegion
AS
BEGIN
    DECLARE @Region NVARCHAR(50), @OrderID INT, @NewFreight DECIMAL(10, 2);

    BEGIN TRANSACTION;

    BEGIN TRY
        DECLARE RegionCursor CURSOR FOR
        SELECT DISTINCT ShipRegion FROM Orders WHERE ShipRegion IS NOT NULL;

        OPEN RegionCursor;
        FETCH NEXT FROM RegionCursor INTO @Region;

        WHILE @@FETCH_STATUS = 0
        BEGIN
            DECLARE OrderCursor CURSOR FOR
            SELECT OrderID, Freight FROM Orders WHERE ShipRegion = @Region;

            OPEN OrderCursor;
            FETCH NEXT FROM OrderCursor INTO @OrderID, @NewFreight;

            WHILE @@FETCH_STATUS = 0
            BEGIN
                -- Exemplo de lógica para ajuste de frete
                SET @NewFreight = @NewFreight * 1.10; 

                UPDATE Orders
                SET Freight = @NewFreight
                WHERE OrderID = @OrderID;

                FETCH NEXT FROM OrderCursor INTO @OrderID, @NewFreight;
            END;

            CLOSE OrderCursor;
            DEALLOCATE OrderCursor;

            FETCH NEXT FROM RegionCursor INTO @Region;
        END;

        CLOSE RegionCursor;
        DEALLOCATE RegionCursor;

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;

        IF CURSOR_STATUS('GLOBAL', 'RegionCursor') >= 0
        BEGIN
            CLOSE RegionCursor;
            DEALLOCATE RegionCursor;
        END;

        IF CURSOR_STATUS('GLOBAL', 'OrderCursor') >= 0
        BEGIN
            CLOSE OrderCursor;
            DEALLOCATE OrderCursor;

        THROW;
```
