### **11. Use uma CTE recursiva para listar a hierarquia de categorias e subcategorias**
Embora o banco Northwind não tenha subcategorias reais, você pode simular uma hierarquia fictícia com base nas categorias existentes.

```sql
WITH HierarquiaCategorias AS (
    SELECT 
        CategoryID,
        CategoryName,
        CAST(CategoryName AS NVARCHAR(MAX)) AS Caminho,
        1 AS Nivel
    FROM Categories
    WHERE ParentCategoryID IS NULL -- Exemplo fictício: subcategoria de categoria pai
    
    UNION ALL
    
    SELECT 
        c.CategoryID,
        c.CategoryName,
        CAST(h.Caminho + ' -> ' + c.CategoryName AS NVARCHAR(MAX)) AS Caminho,
        Nivel + 1
    FROM Categories c
    JOIN HierarquiaCategorias h ON c.ParentCategoryID = h.CategoryID
)
SELECT 
    CategoryID,
    CategoryName,
    Caminho,
    Nivel
FROM HierarquiaCategorias
ORDER BY Caminho;
```

### **14. CTE recursiva para calcular o total acumulado de vendas por ano, começando de 1996**
```sql
WITH VendasAno AS (
    SELECT 
        YEAR(OrderDate) AS Ano,
        SUM(od.Quantity * od.UnitPrice) AS TotalVendas
    FROM Orders o
    JOIN [Order Details] od ON o.OrderID = od.OrderID
    WHERE YEAR(OrderDate) >= 1996
    GROUP BY YEAR(OrderDate)
)
, Acumulado AS (
    SELECT 
        Ano,
        TotalVendas,
        TotalVendas AS Acumulado
    FROM VendasAno
    WHERE Ano = 1996
    
    UNION ALL
    
    SELECT 
        v.Ano,
        v.TotalVendas,
        a.Acumulado + v.TotalVendas AS Acumulado
    FROM VendasAno v
    JOIN Acumulado a ON v.Ano = a.Ano + 1
)
SELECT 
    Ano,
    TotalVendas,
    Acumulado
FROM Acumulado
ORDER BY Ano;
```

### **15. CTE recursiva para calcular o total de produtos vendidos por região e sub-região**
```sql
WITH ProdutosRegiao AS (
    SELECT 
        r.RegionID,
        r.RegionDescription AS Regiao,
        COUNT(DISTINCT p.ProductID) AS TotalProdutos,
        1 AS Nivel
    FROM Region r
    JOIN Territories t ON r.RegionID = t.RegionID
    JOIN EmployeeTerritories et ON t.TerritoryID = et.TerritoryID
    JOIN Orders o ON o.EmployeeID = et.EmployeeID
    JOIN [Order Details] od ON o.OrderID = od.OrderID
    JOIN Products p ON od.ProductID = p.ProductID
    GROUP BY r.RegionID, r.RegionDescription
    
    UNION ALL
    
    SELECT 
        r.RegionID,
        r.RegionDescription,
        pr.TotalProdutos,
        pr.Nivel + 1
    FROM ProdutosRegiao pr
    JOIN Region r ON r.ParentRegionID = pr.RegionID -- Exemplo fictício
)
SELECT 
    Regiao,
    TotalProdutos,
    Nivel
FROM ProdutosRegiao
ORDER BY Nivel, Regiao;
```

### **27. CTE para listar os produtos que tiveram aumento no número de pedidos ano a ano**
```sql
WITH PedidosPorProduto AS (
    SELECT 
        p.ProductID,
        p.ProductName,
        YEAR(o.OrderDate) AS Ano,
        COUNT(o.OrderID) AS TotalPedidos
    FROM Products p
    JOIN [Order Details] od ON p.ProductID = od.ProductID
    JOIN Orders o ON od.OrderID = o.OrderID
    GROUP BY p.ProductID, p.ProductName, YEAR(o.OrderDate)
),
CrescimentoAnoAano AS (
    SELECT 
        p1.ProductID,
        p1.ProductName,
        p1.Ano AS AnoAtual,
        p2.Ano AS AnoAnterior,
        p1.TotalPedidos - p2.TotalPedidos AS Crescimento
    FROM PedidosPorProduto p1
    JOIN PedidosPorProduto p2 ON p1.ProductID = p2.ProductID AND p1.Ano = p2.Ano + 1
)
SELECT 
    ProductID,
    ProductName,
    AnoAtual,
    Crescimento
FROM CrescimentoAnoAano
WHERE Crescimento > 0;
```
