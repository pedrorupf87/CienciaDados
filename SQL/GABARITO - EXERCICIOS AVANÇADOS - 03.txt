-- Gabarito dos Exercícios 01 a 10

-- 01. Consulta dinâmica com transação e log de erros
BEGIN TRY
    BEGIN TRANSACTION;

    DECLARE @sql NVARCHAR(MAX);
    SET @sql = N'
        SELECT OrderID, CustomerID, OrderDate
        FROM Orders
        WHERE CustomerID = @CustomerID
          AND OrderDate BETWEEN @StartDate AND @EndDate';

    EXEC sp_executesql @sql, 
        N'@CustomerID NVARCHAR(5), @StartDate DATETIME, @EndDate DATETIME', 
        @CustomerID = 'ALFKI', @StartDate = '1997-01-01', @EndDate = '1997-12-31';

    COMMIT TRANSACTION;
END TRY
BEGIN CATCH
    ROLLBACK TRANSACTION;
    INSERT INTO ErrorLog (ErrorMessage, ErrorDate)
    VALUES (ERROR_MESSAGE(), GETDATE());
END CATCH;
