### **1. Atualizar o campo `Region` na tabela `Customers` usando um cursor**
```sql
DECLARE @CustomerID NVARCHAR(5), @Region NVARCHAR(15);

DECLARE customer_cursor CURSOR FOR
SELECT CustomerID, Region
FROM Customers;

OPEN customer_cursor;

FETCH NEXT FROM customer_cursor INTO @CustomerID, @Region;

WHILE @@FETCH_STATUS = 0
BEGIN
    UPDATE Customers
    SET Region = ISNULL(@Region, 'N/A')
    WHERE CustomerID = @CustomerID;
    
    FETCH NEXT FROM customer_cursor INTO @CustomerID, @Region;
END;

CLOSE customer_cursor;
DEALLOCATE customer_cursor;
```

### **34. Inserir pedidos de XML e registrar erros em JSON**
```sql
DECLARE @OrderXML XML = 
    '<Orders>
        <Order>
            <CustomerID>ALFKI</CustomerID>
            <OrderDate>2025-01-01</OrderDate>
            <Details>
                <ProductID>1</ProductID>
                <Quantity>5</Quantity>
                <UnitPrice>20.0</UnitPrice>
            </Details>
            <Details>
                <ProductID>999</ProductID> <!-- Produto inexistente -->
                <Quantity>3</Quantity>
                <UnitPrice>15.0</UnitPrice>
            </Details>
        </Order>
    </Orders>';

DECLARE @Log NVARCHAR(MAX) = '[]';
DECLARE @OrderID INT, @CustomerID NVARCHAR(5), @OrderDate DATE;

DECLARE order_cursor CURSOR FOR
SELECT 
    C.value('(CustomerID)[1]', 'NVARCHAR(5)') AS CustomerID,
    C.value('(OrderDate)[1]', 'DATE') AS OrderDate
FROM @OrderXML.nodes('/Orders/Order') AS T(C);

OPEN order_cursor;
FETCH NEXT FROM order_cursor INTO @CustomerID, @OrderDate;

WHILE @@FETCH_STATUS = 0
BEGIN
    BEGIN TRY
        -- Inserir pedido
        INSERT INTO Orders (CustomerID, OrderDate) 
        VALUES (@CustomerID, @OrderDate);

        SET @OrderID = SCOPE_IDENTITY();

        -- Processar detalhes
        INSERT INTO [Order Details] (OrderID, ProductID, Quantity, UnitPrice)
        SELECT 
            @OrderID,
            D.value('(ProductID)[1]', 'INT'),
            D.value('(Quantity)[1]', 'INT'),
            D.value('(UnitPrice)[1]', 'MONEY')
        FROM @OrderXML.nodes('/Orders/Order/Details') AS T(D);
    END TRY
    BEGIN CATCH
        -- Registrar erros
        SET @Log = JSON_MODIFY(
            @Log, 
            'append $', 
            JSON_QUERY(CONCAT(
                '{"CustomerID":"', @CustomerID, 
                '","Error":"', ERROR_MESSAGE(), '"}'))
        );
    END CATCH;

    FETCH NEXT FROM order_cursor INTO @CustomerID, @OrderDate;
END;

CLOSE order_cursor;
DEALLOCATE order_cursor;

-- Log de erros
SELECT @Log AS ErrorLog;
```

```sql
            <NewStock>50</NewStock>
        </Product>
        <Product>
            <ProductID>2</ProductID>
            <NewStock>100</NewStock>
        </Product>
    </Adjustments>';

DECLARE @ProductID INT, @NewStock INT;
DECLARE @Log NVARCHAR(MAX) = '[]';

DECLARE stock_cursor CURSOR FOR
SELECT 
    C.value('(ProductID)[1]', 'INT') AS ProductID,
    C.value('(NewStock)[1]', 'INT') AS NewStock
FROM @StockAdjustments.nodes('/Adjustments/Product') AS T(C);

OPEN stock_cursor;
FETCH NEXT FROM stock_cursor INTO @ProductID, @NewStock;

WHILE @@FETCH_STATUS = 0
BEGIN
    DECLARE @OldStock INT;
    SELECT @OldStock = UnitsInStock FROM Products WHERE ProductID = @ProductID;

    UPDATE Products
    SET UnitsInStock = @NewStock
    WHERE ProductID = @ProductID;

    -- Registrar ajuste
    SET @Log = JSON_MODIFY(
        @Log, 
        'append $', 
        JSON_QUERY(CONCAT(
            '{"ProductID":', @ProductID, 
            ',"OldStock":', @OldStock, 
            ',"NewStock":', @NewStock, '}'))
    );

    FETCH NEXT FROM stock_cursor INTO @ProductID, @NewStock;
END;

CLOSE stock_cursor;
DEALLOCATE stock_cursor;

-- Log de ajustes
SELECT @Log AS StockAdjustmentLog;
```

### **36. Sincronizar dados entre JSON e XML em uma única execução**

```sql
WITH (
```

### **37. Converter XML com detalhes de pedidos para JSON agrupado por cliente**
```sql
DECLARE @OrderXML XML = 
    '<Orders>
        <Order>
            <CustomerID>ALFKI</CustomerID>
            <OrderID>10248</OrderID>
            <ProductID>1</ProductID>
            <Quantity>10</Quantity>
        </Order>
        <Order>
            <CustomerID>ALFKI</CustomerID>
            <OrderID>10248</OrderID>
            <ProductID>2</ProductID>
            <Quantity>5</Quantity>
        </Order>
        <Order>
            <CustomerID>ANATR</CustomerID>
            <OrderID>10249</OrderID>
            <ProductID>1</ProductID>
            <Quantity>3</Quantity>
        </Order>
    </Orders>';

SELECT 
    CustomerID,
    JSON_QUERY((
        SELECT 
            OrderID,
            JSON_QUERY((
                SELECT ProductID, Quantity
                FROM (
                    SELECT 
                        O.value('(ProductID)[1]', 'INT') AS ProductID,
                        O.value('(Quantity)[1]', 'INT') AS Quantity
                    FROM @OrderXML.nodes('/Orders/Order') AS Sub(O)
                    WHERE Sub.O.value('(OrderID)[1]', 'INT') = Orders.OrderID
                ) AS Products
                FOR JSON AUTO
            )) AS Products
        FROM (
            SELECT DISTINCT 
                O.value('(OrderID)[1]', 'INT') AS OrderID
            FROM @OrderXML.nodes('/Orders/Order') AS Sub(O)
        ) AS Orders
        FOR JSON AUTO
    )) AS Orders
FROM (
    SELECT DISTINCT 
        O.value('(CustomerID)[1]', 'NVARCHAR(5)') AS CustomerID
    FROM @OrderXML.nodes('/Orders/Order') AS O
) AS Customers
FOR JSON AUTO;
```

### **38. Validar e corrigir inconsistências entre JSON e XML usando cursores**
```sql
DECLARE @ProductJSON NVARCHAR(MAX) = 
    '[{"ProductID":1,"ProductName":"Chai","Stock":39},
      {"ProductID":2,"ProductName":"Chang","Stock":17}]';

DECLARE @ProductXML XML = 
    '<Products>
        <Product>
            <ProductID>1</ProductID>
            <ProductName>Chai</ProductName>
            <Stock>40</Stock>
        </Product>
        <Product>
            <ProductID>3</ProductID>
            <ProductName>New Product</ProductName>
            <Stock>25</Stock>
        </Product>
    </Products>';

DECLARE @ProductID INT, @JSONStock INT, @XMLStock INT;

DECLARE product_cursor CURSOR FOR
SELECT 
    JSONData.ProductID,
    JSONData.Stock AS JSONStock,
    XMLData.Stock AS XMLStock
FROM 
    OPENJSON(@ProductJSON)
    WITH (
        ProductID INT '$.ProductID',
        ProductName NVARCHAR(50) '$.ProductName',
        Stock INT '$.Stock'
    ) AS JSONData
LEFT JOIN (
    SELECT 
        C.value('(ProductID)[1]', 'INT') AS ProductID,
        C.value('(Stock)[1]', 'INT') AS Stock
    FROM @ProductXML.nodes('/Products/Product') AS T(C)
) AS XMLData
ON JSONData.ProductID = XMLData.ProductID;

OPEN product_cursor;
FETCH NEXT FROM product_cursor INTO @ProductID, @JSONStock, @XMLStock;

WHILE @@FETCH_STATUS = 0
BEGIN
    IF @JSONStock IS NULL OR @XMLStock IS NULL OR @JSONStock <> @XMLStock
    BEGIN
        -- Corrigir inconsistências
        UPDATE Products
        SET UnitsInStock = ISNULL(@XMLStock, @JSONStock)
        WHERE ProductID = @ProductID;
    END;

    FETCH NEXT FROM product_cursor INTO @ProductID, @JSONStock, @XMLStock;
END;

CLOSE product_cursor;
DEALLOCATE product_cursor;
```

### **39. Criar log de operações de atualização em JSON e consolidar em XML**
```sql
DECLARE @Log NVARCHAR(MAX) = 
    '[{"Operation":"Update","Table":"Products","ProductID":1,"OldValue":39,"NewValue":40},
      {"Operation":"Insert","Table":"Orders","OrderID":10250}]';

SELECT 
    Operation,
    (
        SELECT TableName, Details
        FROM (
            SELECT 
                JSON_VALUE(Value, '$.Table') AS TableName,
                JSON_QUERY(Value, '$') AS Details
            FROM OPENJSON(@Log)
        ) AS Operations
        FOR XML PATH('Operation'), TYPE
    ) AS OperationDetails
FROM (
    SELECT DISTINCT JSON_VALUE(Value, '$.Operation') AS Operation
    FROM OPENJSON(@Log)
) AS LogData
FOR XML PATH('Operations'), ROOT('Log');
```
