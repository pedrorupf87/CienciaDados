### **Gabarito dos Exercícios sobre Cursores**

#### **1. Iterar sobre Pedidos**  
```sql
DECLARE @Country NVARCHAR(50), @FreightTotal MONEY;

CREATE TABLE #CountryFreight (Country NVARCHAR(50), FreightTotal MONEY);

DECLARE CursorOrders CURSOR FOR  
SELECT ShipCountry, SUM(Freight)  
FROM Orders  
GROUP BY ShipCountry;

OPEN CursorOrders;

FETCH NEXT FROM CursorOrders INTO @Country, @FreightTotal;

WHILE @@FETCH_STATUS = 0  
BEGIN  
    INSERT INTO #CountryFreight VALUES (@Country, @FreightTotal);  
    FETCH NEXT FROM CursorOrders INTO @Country, @FreightTotal;  
END;

CLOSE CursorOrders;  
DEALLOCATE CursorOrders;

SELECT * FROM #CountryFreight;
DROP TABLE #CountryFreight;
```

#### **9. Histórico de Pedidos**
```sql
DECLARE @CustomerID NCHAR(5), @OrderID INT, @OrderYear INT;

CREATE TABLE #CustomerOrderHistory (CustomerID NCHAR(5), OrderID INT, OrderYear INT);

DECLARE CursorCustomers CURSOR FOR  
SELECT DISTINCT CustomerID  
FROM Orders;

OPEN CursorCustomers;

FETCH NEXT FROM CursorCustomers INTO @CustomerID;

WHILE @@FETCH_STATUS = 0  
BEGIN  
    DECLARE CursorOrders CURSOR FOR  
    SELECT OrderID, YEAR(OrderDate) AS OrderYear  
    FROM Orders  
    WHERE CustomerID = @CustomerID;

    OPEN CursorOrders;

    FETCH NEXT FROM CursorOrders INTO @OrderID, @OrderYear;

    WHILE @@FETCH_STATUS = 0  
    BEGIN  
        INSERT INTO #CustomerOrderHistory VALUES (@CustomerID, @OrderID, @OrderYear);  
        FETCH NEXT FROM CursorOrders INTO @OrderID, @OrderYear;  
    END;

    CLOSE CursorOrders;  
    DEALLOCATE CursorOrders;

    FETCH NEXT FROM CursorCustomers INTO @CustomerID;  
END;

CLOSE CursorCustomers;  
DEALLOCATE CursorCustomers;

SELECT * FROM #CustomerOrderHistory;
DROP TABLE #CustomerOrderHistory;
```

#### **14. Análise de Frete**
```sql
DECLARE @OrderID INT, @Freight MONEY, @OrderTotal MONEY, @FreightPercentage FLOAT;

CREATE TABLE #FreightAnalysis (OrderID INT, FreightPercentage FLOAT);

DECLARE CursorOrders CURSOR FOR  
SELECT OrderID, Freight,  
       (SELECT SUM(UnitPrice * Quantity) FROM [Order Details] WHERE OrderID = O.OrderID) AS OrderTotal  
FROM Orders O;

OPEN CursorOrders;

FETCH NEXT FROM CursorOrders INTO @OrderID, @Freight, @OrderTotal;

WHILE @@FETCH_STATUS = 0  
BEGIN  
    SET @FreightPercentage = (@Freight / @OrderTotal) * 100;  

    INSERT INTO #FreightAnalysis VALUES (@OrderID, @FreightPercentage);  

    FETCH NEXT FROM CursorOrders INTO @OrderID, @Freight, @OrderTotal;  
END;

CLOSE CursorOrders;  
DEALLOCATE CursorOrders;

SELECT * FROM #FreightAnalysis;
DROP TABLE #FreightAnalysis;
```

#### **15. Histórico de Clientes**
```sql
DECLARE @CustomerID NCHAR(5), @ProductID INT;

CREATE TABLE #CustomerProductHistory (CustomerID NCHAR(5), ProductID INT);

DECLARE CursorCustomers CURSOR FOR  
SELECT DISTINCT CustomerID  
FROM Orders;

OPEN CursorCustomers;

FETCH NEXT FROM CursorCustomers INTO @CustomerID;

WHILE @@FETCH_STATUS = 0  
BEGIN  
    DECLARE CursorProducts CURSOR FOR  
    SELECT TOP 1 ProductID  
    FROM [Order Details] OD  
    INNER JOIN Orders O ON OD.OrderID = O.OrderID  
    WHERE O.CustomerID = @CustomerID  
    GROUP BY ProductID  
    ORDER BY SUM(Quantity) DESC;

    OPEN CursorProducts;

    FETCH NEXT FROM CursorProducts INTO @ProductID;

    WHILE @@FETCH_STATUS = 0  
    BEGIN  
        INSERT INTO #CustomerProductHistory VALUES (@CustomerID, @ProductID);  
        FETCH NEXT FROM CursorProducts INTO @ProductID;  
    END;

    CLOSE CursorProducts;  
    DEALLOCATE CursorProducts;

    FETCH NEXT FROM CursorCustomers INTO @CustomerID;  
END;

CLOSE CursorCustomers;  
DEALLOCATE CursorCustomers;

SELECT * FROM #CustomerProductHistory;
DROP TABLE #CustomerProductHistory;
```

    DECLARE CursorProducts CURSOR FOR  
    SELECT ProductID  
    FROM Products  
    WHERE SupplierID = @SupplierID;

    OPEN CursorProducts;

    FETCH NEXT FROM CursorProducts INTO @ProductID;

    WHILE @@FETCH_STATUS = 0  
    BEGIN  
        DECLARE CursorOrders CURSOR FOR  
        SELECT DISTINCT OrderID  
        FROM [Order Details]  
        WHERE ProductID = @ProductID;

        FETCH NEXT FROM CursorOrders INTO @OrderID;

            INSERT INTO #SupplierOrders VALUES (@SupplierID, @OrderID);  
            FETCH NEXT FROM CursorOrders INTO @OrderID;  
        END;

        CLOSE CursorOrders;  
        DEALLOCATE CursorOrders;

        FETCH NEXT FROM CursorProducts INTO @ProductID;  
    END;

    CLOSE CursorProducts;  
    DEALLOCATE CursorProducts;

    FETCH NEXT FROM CursorSuppliers INTO @SupplierID;  
END;

CLOSE CursorSuppliers;  
DEALLOCATE CursorSuppliers;

SELECT * FROM #SupplierOrders;

#### **19. Análise de Produtos Descontinuados**

FROM Products  
WHERE Discontinued = 1;

WHILE @@FETCH_STATUS = 0

    SET @SalesTotal = (  
        SELECT SUM(UnitPrice * Quantity)  
        FROM [Order Details]
